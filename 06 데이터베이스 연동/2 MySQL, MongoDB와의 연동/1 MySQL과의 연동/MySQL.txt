MySQL과의 연동
MySQL 모듈 설치
- npm install mysql

연결 설정
- MySQL 데이터베이스에 연결하기 위해 필요한 호스트, 사용자명, 비밀번호, 데이터베이스 이름 등의 정보를 설정

const mysql = require('mysql');

const connection = mysql.createConnection({
  host: 'localhost',
  user: 'username',
  password: 'password',
  database: 'database_name',
});

쿼리 실행: 연결 설정이 완료되면 connection.query() 메서드를 사용하여 SQL 쿼리를 실행

connection.query('SELECT * FROM table_name', (error, results, fields) => {
  if (error) throw error;

  // 결과 처리
});
